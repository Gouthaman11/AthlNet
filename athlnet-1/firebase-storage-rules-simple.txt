// SIMPLIFIED Firebase Storage Rules for Testing
// Copy these rules to Firebase Console -> Storage -> Rules

rules_version = '2';

service firebase.storage {
  match /b/{bucket}/o {
    // Allow any authenticated user to read and write anywhere (most permissive)
    match /{allPaths=**} {
      allow read, write: if request.auth != null;
    }
  }
}
